Using the above CONSTRAINT data_is_valid check,  I want to insert one jsonb per validation pattern. 
So the idea is to have a row of jsonbschema as validator. Then from that point onwards the column will follow the afformentioned validation pattern. This should be able to be chunkable with each validation pattern. It is ok to have one extra column in the table to mark a prerequisite changes to the jsonb schema.

    deleted BOOLEAN NOT NULL DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT NOW(),
    updated_by TEXT,
    change_type TEXT,
    history_data JSONB,
    CONSTRAINT postal_address_history_pkey PRIMARY KEY (id, updated_at)

CREATE TRIGGER postal_address_history_updated_by_trigger
    BEFORE INSERT OR UPDATE ON postal_address_history
    FOR EACH ROW
    EXECUTE FUNCTION set_postal_address_updated_by();

CREATE OR REPLACE FUNCTION postal_address_history_trigger_function()
RETURNS TRIGGER AS $$
BEGIN
  IF (TG_OP = 'UPDATE' OR TG_OP = 'DELETE') THEN
    NEW.history_data := jsonb_build_object(
      'postal_address_id', OLD.postal_address_id,
      'deleted', OLD.deleted,
      'created_at', OLD.created_at,
      'updated_by', current_user,
      'change_type', TG_OP
    );
  END IF;

  IF (TG_OP = 'UPDATE' OR TG_OP = 'INSERT') THEN
    NEW.history_data := jsonb_build_object(

      'created_at', NEW.created_at,
      'updated_by', current_user,
      'change_type', TG_OP
    );
  END IF;

END;

CREATE TABLE jsonb_schema_validator_xref (
    id SERIAL PRIMARY KEY,
    validation_pattern TEXT NOT NULL,
    jsonb_schema JSONB NOT NULL
);

CREATE OR REPLACE FUNCTION postal_address_history_trigger_function()
RETURNS TRIGGER AS $$
DECLARE
    schema_record jsonb_schema_validator_xref%ROWTYPE;
    validation_pattern TEXT;
BEGIN
    IF (TG_OP = 'UPDATE' OR TG_OP = 'DELETE') THEN
        validation_pattern := (SELECT validation_pattern FROM jsonb_schema_validator_xref WHERE id = OLD.jsonb_schema);
        schema_record := (SELECT * FROM jsonb_schema_validator_xref WHERE validation_pattern = validation_pattern);
        NEW.history_data := jsonb_build_object(
            'postal_address_id', OLD.postal_address_id,
            'deleted', OLD.deleted,
            'created_at', OLD.created_at,
            'updated_by', current_user,
            'change_type', TG_OP,
            'jsonb_schema', schema_record.jsonb_schema
        );
    END IF;

    IF (TG_OP = 'UPDATE' OR TG_OP = 'INSERT') THEN
        validation_pattern := (SELECT validation_pattern FROM jsonb_schema_validator_xref WHERE id = NEW.jsonb_schema);
        schema_record := (SELECT * FROM jsonb_schema_validator_xref WHERE validation_pattern = validation_pattern);
        NEW.history_data := jsonb_build_object(
            'postal_address_id', NEW.postal_address_id,
            'deleted', NEW.deleted,
            'created_at', NEW.created_at,
            'updated_by', current_user,
            'change_type', TG_OP,
            'jsonb_schema', schema_record.jsonb_schema
        );
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION insert_jsonb_schema_validator(
    p_validation_pattern TEXT,
    p_jsonb_schema JSONB
)
RETURNS VOID AS $$
BEGIN
    INSERT INTO jsonb_schema_validator_xref (validation_pattern, jsonb_schema)
    VALUES (p_validation_pattern, p_jsonb_schema);

    -- Create the trigger to execute the ALTER TABLE statement on each insert
    EXECUTE '
        CREATE OR REPLACE FUNCTION data_is_valid_trigger_function()
        RETURNS TRIGGER AS $$
        DECLARE
            v_jsonb_schema JSONB;
        BEGIN
            SELECT jsonb_schema INTO v_jsonb_schema
            FROM jsonb_schema_validator_xref;
        
            EXECUTE
                ''ALTER TABLE postal_address ADD CONSTRAINT data_is_valid CHECK (
                    validate_json_schema('''' || v_jsonb_schema || '''', NEW.data)
                )'';
        
            RETURN NEW;
        END;
        $$ LANGUAGE plpgsql;
        
        CREATE TRIGGER data_is_valid_trigger
        AFTER INSERT ON jsonb_schema_validator_xref
        FOR EACH ROW
        EXECUTE FUNCTION data_is_valid_trigger_function()
    ';
END;
$$ LANGUAGE plpgsql;
